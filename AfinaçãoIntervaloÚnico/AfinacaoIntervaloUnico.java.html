import java.util.*;

public class AfinacaoIntervaloUnico {
    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        System.out.print("Digite uma frequência fundamental (ex: 440): ");
        double a4 = scanner.nextDouble();

        if(a4 <= 50){
            a4 *= 2;
        }
        else if(a4 >= 120 && a4 <= 220){
            a4 /= 2;
        }
        else if(a4 >= 221 && a4 <= 460){
            a4 /= 4;
        }
        else if(a4 >= 461 && a4 <= 880){
            a4 /= 8;
        }
        else if(a4 >= 881 && a4 <= 1760){
            a4 /= 16;
        }

        System.out.print("Digite um numerador: ");
        int numerador = scanner.nextInt();

        System.out.print("Digite um denominador: ");
        int denominador = scanner.nextInt();

        double frequencia = a4;

        // Ordem FIXA das notas
        String[] ordemNotas = {
                "A", "A#", "B", "C", "C#", "D", "D#", "E", "F", "F#", "G", "G#"
        };

        // Map final
        Map<String, Double> notas = new HashMap<>();

        // Lista só de frequências (para ordenar)
        List<Double> frequencias = new ArrayList<>();

        // ===== GERA AS FREQUÊNCIAS =====
        for (int i = 0; i < 12; i++) {

            if (i > 0) {
                frequencia = frequencia * numerador / denominador; // sua lógica original
            }

            while (frequencia >= a4 * 2) {
                frequencia /= 2.0;
            }
            while (frequencia < a4) {
                frequencia *= 2.0;
            }

            frequencias.add(frequencia);
        }

        // ===== ORDENA DO MENOR PARA O MAIOR =====
        Collections.sort(frequencias);

        // ===== ATRIBUI NA ORDEM FIXA DE NOTAS =====
        for (int i = 0; i < 12; i++) {
            notas.put(ordemNotas[i], frequencias.get(i));
        }

        // ===== MOSTRAR NA ORDEM FIXA =====
        System.out.println("\nNotas ordenadas:\n");
        for (String nota : ordemNotas) {
            System.out.printf("%s = %.3f Hz\n", nota, notas.get(nota));
        }

        System.out.println("-----------------");
        System.out.println("Afinação violão: ");
        System.out.println("-----------------");

        // ===== AFINAÇÃO PADRÃO BASEADA NA SUA ESCALA GERADA =====
        Map<String, Double> afinacaoGuitarra = new LinkedHashMap<>();

        afinacaoGuitarra.put("E2", notas.get("E") / 2);
        afinacaoGuitarra.put("A2", notas.get("A"));
        afinacaoGuitarra.put("D3", notas.get("D"));
        afinacaoGuitarra.put("G3", notas.get("G"));
        afinacaoGuitarra.put("B3", notas.get("B") * 2);
        afinacaoGuitarra.put("E4", notas.get("E") * 2); // oitava acima

        System.out.println("\nAfinação de violão/guitarra baseada na sua escala:");
        for (var e : afinacaoGuitarra.entrySet()) {
            System.out.printf("%s = %.6f Hz\n", e.getKey(), e.getValue());
        }

    }
}
