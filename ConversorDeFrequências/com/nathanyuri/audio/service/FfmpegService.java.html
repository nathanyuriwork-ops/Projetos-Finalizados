package com.nathanyuri.audio.service;

import com.nathanyuri.audio.model.ConversionRequest;
import com.nathanyuri.audio.util.FfmpegUtils;

import java.io.IOException;
import java.util.Arrays;
import java.util.List;

public class FfmpegService {

    public void convert(ConversionRequest request) {

        double ratio = request.getTargetHz()
                / request.getOriginalHz();

        double gainDb = request.getGainDb(); // ex: +3.0

        String filter = String.format(
                "aresample=96000," +
                        "rubberband=pitch=%.10f:tempo=1.0:formant=1:transients=crisp:window=long," +
                        "volume=%.1fdB," +
                        "aresample=48000",
                ratio,
                gainDb
        );


        List<String> command = Arrays.asList(
                "ffmpeg",
                "-y",
                "-i", request.getInputPath(),
                "-af", filter,
                "-acodec", "pcm_f32le",
                request.getOutputPath()
        );


        ProcessBuilder processBuilder = new ProcessBuilder(command);
        processBuilder.inheritIO();

        try {
            Process process = processBuilder.start();
            int exitCode = process.waitFor();

            if (exitCode != 0) {
                throw new RuntimeException("Erro ao executar FFmpeg. Código: " + exitCode);
            }

        } catch (IOException | InterruptedException e) {
            throw new RuntimeException("Falha na conversão de áudio", e);
        }
    }
}
