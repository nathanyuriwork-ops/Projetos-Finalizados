package com.nathanyuri.audio.util;

public class FfmpegUtils {

    public static String buildPitchFilter(double originalHz, double targetHz) {

        double ratio = (double) targetHz / originalHz;
        double tempoCompensation = 1.0 / ratio;

        return String.format(
                "asetrate=44100*%.8f,aresample=44100,atempo=%.8f",
                ratio,
                tempoCompensation
        );
    }
}
