package com.nathanyuri.audio.view;

import com.nathanyuri.audio.controller.FxController;
import javafx.geometry.Insets;
import javafx.scene.Parent;
import javafx.scene.control.*;
import javafx.scene.layout.GridPane;

public class MainView {

    private final FxController controller = new FxController();

    private TextField fileField;
    private TextField originalHzField;
    private TextField targetHzField;
    private Label statusLabel;

    public Parent getRoot() {

        GridPane grid = new GridPane();
        grid.setPadding(new Insets(20));
        grid.setHgap(10);
        grid.setVgap(10);

        fileField = new TextField();
        fileField.setEditable(false);

        Button browseButton = new Button("Selecionar arquivo");
        browseButton.setOnAction(e -> controller.chooseFile(fileField));

        originalHzField = new TextField("440");
        targetHzField = new TextField("432");

        Button convertButton = new Button("Converter");
        convertButton.setOnAction(e ->
                controller.convert(
                        fileField.getText(),
                        originalHzField.getText(),
                        targetHzField.getText(),
                        statusLabel
                )
        );

        statusLabel = new Label("Aguardando ação...");

        grid.add(new Label("Arquivo:"), 0, 0);
        grid.add(fileField, 1, 0);
        grid.add(browseButton, 2, 0);

        grid.add(new Label("Hz original:"), 0, 1);
        grid.add(originalHzField, 1, 1);

        grid.add(new Label("Hz destino:"), 0, 2);
        grid.add(targetHzField, 1, 2);

        grid.add(convertButton, 1, 3);
        grid.add(statusLabel, 1, 4);

        return grid;
    }
}
