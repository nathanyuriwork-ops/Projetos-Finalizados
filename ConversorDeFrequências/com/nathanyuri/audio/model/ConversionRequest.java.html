package com.nathanyuri.audio.model;

public class ConversionRequest {

    private final String inputPath;
    private final String outputPath;
    private final double originalFrequency;
    private final double targetFrequency;
    private final double gainDb;

    public ConversionRequest(String inputPath,
                             String outputPath,
                             double originalFrequency,
                             double targetFrequency,
                             double gainDb) {
        this.inputPath = inputPath;
        this.outputPath = outputPath;
        this.originalFrequency = originalFrequency;
        this.targetFrequency = targetFrequency;
        this.gainDb = gainDb;
    }

    public String getInputPath() {
        return inputPath;
    }

    public String getOutputPath() {
        return outputPath;
    }

    public double getOriginalHz() {
        return originalFrequency;
    }

    public double getTargetHz() {
        return targetFrequency;
    }

    public double getGainDb(){
        return gainDb;
    }
}
